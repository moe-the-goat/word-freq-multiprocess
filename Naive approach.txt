#include <sys/time.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <unistd.h>

#define MAX_WORD_LENGTH 70

// Define the structure to store words and their frequencies
typedef struct {
    char word[MAX_WORD_LENGTH];
    int frequency;
} WordArray;

// Function to sort words based on frequency using linear sort
void linearSort(WordArray *words, int wordCount) {
    for (int i = 0; i < wordCount - 1; i++) {
        for (int j = i + 1; j < wordCount; j++) {
            if (words[j].frequency > words[i].frequency) {
                // Swap the words
                WordArray temp = words[i];
                words[i] = words[j];
                words[j] = temp;
            }
        }
    }
}

// Function to count word frequencies in the given file
void countFrequency(const char *fileName) {
    WordArray *words = NULL;
    int wordCount = 0;
    int capacity = 10;  // Initial capacity
    char buffer[MAX_WORD_LENGTH];

    // Allocate initial memory
    words = malloc(capacity * sizeof(WordArray));
    if (!words) {
        perror("Error allocating memory");
        exit(1);
    }

    // Attempt to open the file
    FILE *file = fopen(fileName, "r");
    if (!file) {
        perror("Error opening file");
        free(words);  // Free allocated memory before exiting
        exit(1);
    }

    // Read words from the file and count their frequencies
    while (fscanf(file, "%69s", buffer) != EOF) {
        int found = 0;
        for (int i = 0; i < wordCount; i++) {
            if (strcmp(words[i].word, buffer) == 0) {
                words[i].frequency++;
                found = 1;
                break;
            }
        }

        if (!found) {
            // If the array is full, double its capacity
            if (wordCount == capacity) {
                capacity *= 2;
                WordArray *temp = realloc(words, capacity * sizeof(WordArray));
                if (!temp) {
                    perror("Error reallocating memory");
                    free(words);  // Free allocated memory before exiting
                    fclose(file);
                    exit(1);
                }
                words = temp;
            }

            // Add the new word
            strncpy(words[wordCount].word, buffer, MAX_WORD_LENGTH - 1);
            words[wordCount].word[MAX_WORD_LENGTH - 1] = '\0';  // Null-terminate
            words[wordCount].frequency = 1;
            wordCount++;
        }
    }

    fclose(file);

    // Sort and print the top 10 most frequent words
    linearSort(words, wordCount);
    printf("Top 10 Most Frequent Words:\n");
    for (int i = 0; i < 10 && i < wordCount; i++) {
        printf("%s: %d\n", words[i].word, words[i].frequency);
    }

    free(words);  // Free allocated memory
}

int main(int argc, char *argv[]) {
    struct timeval start, end;
    double duration;

    gettimeofday(&start, NULL);
    if (argc != 2) {
        fprintf(stderr, "Usage: %s <file_name>\n", argv[0]);
        return 1;  // Exit code 1 for incorrect usage
    }

    countFrequency(argv[1]);

    gettimeofday(&end, NULL);
    duration = (end.tv_sec - start.tv_sec) * 1000000.0;
    duration += (end.tv_usec - start.tv_usec);
    printf("\n%f minutes\n", duration / 60000000.0);
    return 0;
}
